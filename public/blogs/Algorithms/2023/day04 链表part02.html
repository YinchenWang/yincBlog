    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        ListNode* curA = headA;
        ListNode* curB = headB;
        int lenA = 0;
        int lenB = 0;
        while(curA != NULL)
        {
            lenA++;
            curA = curA -> next;
        }

        while(curB != NULL)
        {
            lenB++;
            curB = curB -> next;
        }

        curB = headB;
